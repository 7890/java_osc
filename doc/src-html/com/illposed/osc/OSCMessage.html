009package com.illposed.osc;
010
011import com.illposed.osc.utility.OSCJavaToByteArrayConverter;
012import java.util.ArrayList;
013import java.util.Collection;
014import java.util.Collections;
015import java.util.LinkedList;
016import java.util.List;
017import java.util.regex.Pattern;
018
019/**
020 * An simple (non-bundle) OSC message.
021 *
022 * An OSC <i>Message</i> is made up of
023 * an <i>Address Pattern</i> (the receiver of the message)
024 * and <i>Arguments</i> (the content of the message).
025 *
026 * @author Chandrasekhar Ramakrishnan
027 */
028public class OSCMessage extends AbstractOSCPacket {
029
030  /**
031   * Java regular expression pattern matching a single invalid character.
032   * The invalid characters are:
033   * ' ', '#', '*', ',', '?', '[', ']', '{', '}'
034   */
035  private static final Pattern ILLEGAL_ADDRESS_CHAR
036      = Pattern.compile("[ \\#\\*\\,\\?\\[\\]\\{\\}]");
037
038  private String address;
039  private List<Object> arguments;
040
041  /**
042   * Creates an empty OSC Message.
043   * In order to send this OSC message,
044   * you need to set the address and optionally some arguments.
045   */
046  public OSCMessage() {
047    this(null);
048  }
049
050  /**
051   * Creates an OSCMessage with an address already initialized.
052   * @param address  the recipient of this OSC message
053   */
054  public OSCMessage(String address) {
055    this(address, null);
056  }
057
058  /**
059   * Creates an OSCMessage with an address
060   * and arguments already initialized.
061   * @param address  the recipient of this OSC message
062   * @param arguments  the data sent to the receiver
063   */
064  public OSCMessage(String address, Collection<Object> arguments) {
065
066    checkAddress(address);
067    this.address = address;
068    if (arguments == null) {
069      this.arguments = new LinkedList<Object>();
070    } else {
071      this.arguments = new ArrayList<Object>(arguments);
072    }
073  }
074
075  /**
076   * The receiver of this message.
077   * @return the receiver of this OSC Message
078   */
079  public String getAddress() {
080    return address;
081  }
082
083  /**
084   * Set the address of this message.
085   * @param address the receiver of the message
086   */
087  public void setAddress(String address) {
088    checkAddress(address);
089    this.address = address;
090    contentChanged();
091  }
092
093  /**
094   * Add an argument to the list of arguments.
095   * @param argument a Float, Double, String, Character, Integer, Long, Boolean, null
096   *   or an array of these
097   */
098  public void addArgument(Object argument) {
099    arguments.add(argument);
100    contentChanged();
101  }
102
103  /**
104   * The arguments of this message.
105   * @return the arguments to this message
106   */
107  public List<Object> getArguments() {
108    return Collections.unmodifiableList(arguments);
109  }
110
111  /**
112   * Convert the address into a byte array.
113   * Used internally only.
114   * @param stream where to write the address to
115   */
116  private void computeAddressByteArray(OSCJavaToByteArrayConverter stream) {
117    stream.write(address);
118  }
119
120  /**
121   * Convert the arguments into a byte array.
122   * Used internally only.
123   * @param stream where to write the arguments to
124   */
125  private void computeArgumentsByteArray(OSCJavaToByteArrayConverter stream) {
126    stream.write(',');
127    stream.writeTypes(arguments);
128    for (final Object argument : arguments) {
129      stream.write(argument);
130    }
131  }
132
133  @Override
134  protected byte[] computeByteArray(OSCJavaToByteArrayConverter stream) {
135    computeAddressByteArray(stream);
136    computeArgumentsByteArray(stream);
137    return stream.toByteArray();
138  }
139
140  /**
141   * Throws an exception if the given address is invalid.
142   * We explicitly allow <code>null</code> here,
143   * because we want to allow to set the address in a lazy fashion.
144   * @param address to be checked for validity
145   */
146  private static void checkAddress(String address) {
147    // NOTE We explicitly allow <code>null</code> here,
148    //   because we want to allow to set in a lazy fashion.
149    if ((address != null) && !isValidAddress(address)) {
150      throw new IllegalArgumentException("Not a valid OSC address: " + address);
151    }
152  }
153
154  /**
155   * Checks whether a given string is a valid OSC <i>Address Pattern</i>.
156   * @param address to be checked for validity
157   * @return true if the supplied string constitutes a valid OSC address
158   */
159  public static boolean isValidAddress(String address) {
160    return (address != null)
161        && !address.isEmpty()
162        && address.charAt(0) == '/'
163        && !address.contains("//")
164        && !ILLEGAL_ADDRESS_CHAR.matcher(address).find();
165  }
166}
