009package com.illposed.osc;
010
011import com.illposed.osc.utility.OSCJavaToByteArrayConverter;
012import java.util.ArrayList;
013import java.util.Collection;
014import java.util.Collections;
015import java.util.Date;
016import java.util.LinkedList;
017import java.util.List;
018
019/**
020 * A bundle represents a collection of OSC packets
021 * (either messages or other bundles)
022 * and has a time-tag which can be used by a scheduler to execute
023 * a bundle in the future,
024 * instead of immediately.
025 * {@link OSCMessage}s are executed immediately.
026 *
027 * Bundles should be used if you want to send multiple messages to be executed
028 * atomically together, or you want to schedule one or more messages to be
029 * executed in the future.
030 *
031 * @author Chandrasekhar Ramakrishnan
032 */
033public class OSCBundle extends AbstractOSCPacket {
034
035  /**
036   * 2208988800 seconds -- includes 17 leap years
037   */
038  public static final long SECONDS_FROM_1900_TO_1970 = 2208988800L;
039
040  /**
041   * The Java representation of an OSC timestamp with the semantics of
042   * "immediately".
043   */
044  public static final Date TIMESTAMP_IMMEDIATE = new Date(0);
045
046  private Date timestamp;
047  private List<OSCPacket> packets;
048
049  /**
050   * Create a new empty OSCBundle with a timestamp of immediately.
051   * You can add packets to the bundle with addPacket()
052   */
053  public OSCBundle() {
054    this(TIMESTAMP_IMMEDIATE);
055  }
056
057  /**
058   * Create an OSCBundle with the specified timestamp.
059   * @param timestamp the time to execute the bundle
060   */
061  public OSCBundle(Date timestamp) {
062    this(null, timestamp);
063  }
064
065  /**
066   * Creates an OSCBundle made up of the given packets
067   * with a timestamp of now.
068   * @param packets array of OSCPackets to initialize this object with
069   */
070  public OSCBundle(Collection<OSCPacket> packets) {
071    this(packets, TIMESTAMP_IMMEDIATE);
072  }
073
074  /**
075   * Create an OSCBundle, specifying the packets and timestamp.
076   * @param packets the packets that make up the bundle
077   * @param timestamp the time to execute the bundle
078   */
079  public OSCBundle(Collection<OSCPacket> packets, Date timestamp) {
080
081    if (null == packets) {
082      this.packets = new LinkedList<OSCPacket>();
083    } else {
084      this.packets = new ArrayList<OSCPacket>(packets);
085    }
086    this.timestamp = clone(timestamp);
087  }
088
089  private static Date clone(final Date toBeCloned) {
090    return (toBeCloned == null) ? toBeCloned : (Date) toBeCloned.clone();
091  }
092
093  /**
094   * Return the time the bundle will execute.
095   * @return a Date
096   */
097  public Date getTimestamp() {
098    return clone(timestamp);
099  }
100
101  /**
102   * Set the time the bundle will execute.
103   * @param timestamp Date
104   */
105  public void setTimestamp(Date timestamp) {
106    this.timestamp = clone(timestamp);
107  }
108
109  /**
110   * Add a packet to the list of packets in this bundle.
111   * @param packet OSCMessage or OSCBundle
112   */
113  public void addPacket(OSCPacket packet) {
114    packets.add(packet);
115    contentChanged();
116  }
117
118  /**
119   * Get the packets contained in this bundle.
120   * @return the packets contained in this bundle.
121   */
122  public List<OSCPacket> getPackets() {
123    return Collections.unmodifiableList(packets);
124  }
125
126  /**
127   * Convert the time-tag (a Java Date) into the OSC byte stream.
128   * Used Internally.
129   * @param stream where to write the time-tag to
130   */
131  private void computeTimeTagByteArray(OSCJavaToByteArrayConverter stream) {
132    if ((null == timestamp) || (timestamp.equals(TIMESTAMP_IMMEDIATE))) {
133      stream.write((int) 0);
134      stream.write((int) 1);
135      return;
136    }
137
138    final long millisecs = timestamp.getTime();
139    final long secsSince1970 = (long) (millisecs / 1000);
140    final long secs = secsSince1970 + SECONDS_FROM_1900_TO_1970;
141
142    // this line was cribbed from jakarta commons-net's NTP TimeStamp code
143    final long fraction = ((millisecs % 1000) * 0x100000000L) / 1000;
144
145    stream.write((int) secs);
146    stream.write((int) fraction);
147  }
148
149  @Override
150  protected byte[] computeByteArray(OSCJavaToByteArrayConverter stream) {
151    stream.write("#bundle");
152    computeTimeTagByteArray(stream);
153    byte[] packetBytes;
154    for (final OSCPacket pkg : packets) {
155      packetBytes = pkg.getByteArray();
156      stream.write(packetBytes);
157    }
158    return stream.toByteArray();
159  }
160}
