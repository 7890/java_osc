009package com.illposed.osc;
010
011import com.illposed.osc.utility.OSCJavaToByteArrayConverter;
012import java.nio.charset.Charset;
013
014/**
015 * OSCPacket is the abstract superclass for the various
016 * kinds of OSC Messages.
017 *
018 * The actual packets are:
019 * <ul>
020 * <li>{@link OSCMessage}: simple OSC messages
021 * <li>{@link OSCBundle}: OSC messages with timestamps
022 *   and/or made up of multiple messages
023 * </ul>
024 */
025abstract class AbstractOSCPacket implements OSCPacket {
026
027  /** Used to encode message addresses and string parameters. */
028  private Charset charset;
029  private byte[] byteArray;
030
031  public AbstractOSCPacket() {
032    this.charset = Charset.defaultCharset();
033    this.byteArray = null;
034  }
035
036  @Override
037  public Charset getCharset() {
038    return charset;
039  }
040
041  @Override
042  public void setCharset(Charset charset) {
043    this.charset = charset;
044  }
045
046  /**
047   * Generate a representation of this packet conforming to the
048   * the OSC byte stream specification. Used Internally.
049   */
050  private byte[] computeByteArray() {
051    final OSCJavaToByteArrayConverter stream = new OSCJavaToByteArrayConverter();
052    stream.setCharset(charset);
053    return computeByteArray(stream);
054  }
055
056  /**
057   * Produces a byte array representation of this packet.
058   * @param stream where to write the arguments to
059   * @return the OSC specification conform byte array representation
060   *   of this packet
061   */
062  protected abstract byte[] computeByteArray(OSCJavaToByteArrayConverter stream);
063
064  @Override
065  public byte[] getByteArray() {
066    if (byteArray == null) {
067      byteArray = computeByteArray();
068    }
069    return byteArray;
070  }
071
072  protected void contentChanged() {
073    byteArray = null;
074  }
075}
