009package com.illposed.osc;
010
011import java.io.IOException;
012import java.net.DatagramPacket;
013import java.net.DatagramSocket;
014import java.net.InetAddress;
015import java.net.SocketException;
016import java.net.UnknownHostException;
017
018/**
019 * OSCPortOut is the class that sends OSC messages
020 * to a specific address and port.
021 *
022 * To send an OSC message, call {@link #send(OSCPacket)}.
023 *
024 * An example:<br>
025 * (loosely based on {com.illposed.osc.OSCPortTest#testMessageWithArgs()})
026 * <blockquote><pre>{@code
027 * OSCPort sender = new OSCPort();
028 * List<Object> args = new ArrayList<Object>(2);
029 * args.add(3);
030 * args.add("hello");
031 * OSCMessage msg = new OSCMessage("/sayhello", args);
032 * try {
033 *  sender.send(msg);
034 * } catch (Exception ex) {
035 *  showError("Couldn't send");
036 * }
037 * }</pre></blockquote>
038 *
039 * @author Chandrasekhar Ramakrishnan
040 */
041public class OSCPortOut extends OSCPort {
042
043  private InetAddress address;
044
045  /**
046   * Create an OSCPort that sends to address:port using a specified socket.
047   * @param address the UDP address to send to
048   * @param port the UDP port to send to
049   * @param socket the DatagramSocket to send from
050   */
051  public OSCPortOut(InetAddress address, int port, DatagramSocket socket) {
052    super(socket, port);
053    this.address = address;
054  }
055
056  /**
057   * Create an OSCPort that sends to address:port.
058   * @param address the UDP address to send to
059   * @param port the UDP port to send to
060   * @throws SocketException when failing to create a (UDP) out socket
061   */
062  public OSCPortOut(InetAddress address, int port) throws SocketException {
063    this(address, port, new DatagramSocket());
064  }
065
066  /**
067   * Create an OSCPort that sends to address,
068   * using the standard SuperCollider port.
069   * @param address the UDP address to send to
070   * @throws SocketException when failing to create a (UDP) out socket
071   */
072  public OSCPortOut(InetAddress address) throws SocketException {
073    this(address, DEFAULT_SC_OSC_PORT);
074  }
075
076  /**
077   * Create an OSCPort that sends to "localhost",
078   * on the standard SuperCollider port.
079   * @throws UnknownHostException if the local host name could not be resolved into an address
080   * @throws SocketException when failing to create a (UDP) out socket
081   */
082  public OSCPortOut() throws UnknownHostException, SocketException {
083    this(InetAddress.getLocalHost(), DEFAULT_SC_OSC_PORT);
084  }
085
086  /**
087   * Send an OSC packet (message or bundle) to the receiver we are bound to.
088   * @param aPacket the bundle or message to send
089   * @throws IOException if a (UDP) socket I/O error occurs
090   */
091  public void send(OSCPacket aPacket) throws IOException {
092    final byte[] byteArray = aPacket.getByteArray();
093    final DatagramPacket packet =
094        new DatagramPacket(byteArray, byteArray.length, address, getPort());
095    getSocket().send(packet);
096  }
097
098  //set new target address for this port
099  public void setTarget(InetAddress address, int port)
100  {
101    this.address = address;
102    super.setPort(port);
103  }
104}
