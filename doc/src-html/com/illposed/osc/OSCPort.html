009package com.illposed.osc;
010
011import java.net.DatagramSocket;
012
013/**
014 * OSCPort is an abstract superclass, to send OSC messages,
015 * use {@link OSCPortOut}.
016 * To listen for OSC messages, use {@link OSCPortIn}.
017 *
018 * @author Chandrasekhar Ramakrishnan
019 */
020public class OSCPort {
021
022  private final DatagramSocket socket;
023  private int port;
024
025  public static final int DEFAULT_SC_OSC_PORT = 57110;
026  public static final int DEFAULT_SC_LANG_OSC_PORT = 57120;
027
028  protected OSCPort(DatagramSocket socket, int port) {
029    this.socket = socket;
030    this.port = port;
031  }
032
033  /**
034   * The port that the SuperCollider <b>synth</b> engine
035   * usually listens to.
036   * @return default SuperCollider <b>synth</b> UDP port
037   * @see #DEFAULT_SC_OSC_PORT
038   */
039  public static int defaultSCOSCPort() {
040    return DEFAULT_SC_OSC_PORT;
041  }
042
043  /**
044   * The port that the SuperCollider <b>language</b> engine
045   * usually listens to.
046   * @return default SuperCollider <b>language</b> UDP port
047   * @see #DEFAULT_SC_LANG_OSC_PORT
048   */
049  public static int defaultSCLangOSCPort() {
050    return DEFAULT_SC_LANG_OSC_PORT;
051  }
052
053  /**
054   * Returns the socket associated with this port.
055   * @return this ports socket
056   */
057  protected DatagramSocket getSocket() {
058    return socket;
059  }
060
061  /**
062   * Returns the port number associated with this port.
063   * @return this ports number
064   */
065  protected int getPort() {
066    return port;
067  }
068
069  /**
070   * Close the socket and free-up resources.
071   * It is recommended that clients call this when they are done with the
072   * port.
073   */
074  public void close() {
075    socket.close();
076  }
077
078  public void setPort(int port)
079  {
080    this.port=port;
081  }
082}
