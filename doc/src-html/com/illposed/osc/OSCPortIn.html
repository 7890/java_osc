009package com.illposed.osc;
010
011import com.illposed.osc.utility.OSCByteArrayToJavaConverter;
012import com.illposed.osc.utility.OSCPacketDispatcher;
013import com.illposed.osc.utility.OSCPatternAddressSelector;
014import java.io.IOException;
015import java.net.DatagramPacket;
016import java.net.DatagramSocket;
017import java.net.SocketException;
018import java.nio.charset.Charset;
019
020/**
021 * OSCPortIn is the class that listens for OSC messages.
022 *
023 * An example:<br>
024 * (loosely based on {com.illposed.osc.OSCPortTest#testReceiving()})
025 * <blockquote><pre>{@code
026 * receiver = new OSCPortIn(OSCPort.DEFAULT_SC_OSC_PORT());
027 * OSCListener listener = new OSCListener() {
028 *  public void acceptMessage(java.util.Date time, OSCMessage message) {
029 *    System.out.println("Message received!");
030 *  }
031 * };
032 * receiver.addListener("/message/receiving", listener);
033 * receiver.startListening();
034 * }</pre></blockquote>
035 *
036 * Then, using a program such as SuperCollider or sendOSC, send a message
037 * to this computer, port {@link #DEFAULT_SC_OSC_PORT},
038 * with the address "/message/receiving".
039 *
040 * @author Chandrasekhar Ramakrishnan
041 */
042public class OSCPortIn extends OSCPort implements Runnable {
043
044  /**
045   * Buffers were 1500 bytes in size, but were
046   * increased to 1536, as this is a common MTU.
047   */
048  private static final int BUFFER_SIZE = 1536;
049
050  /** state for listening */
051  private boolean listening;
052  private final OSCByteArrayToJavaConverter converter;
053  private final OSCPacketDispatcher dispatcher;
054
055  /**
056   * Create an OSCPort that listens using a specified socket.
057   * @param socket DatagramSocket to listen on.
058   */
059  public OSCPortIn(DatagramSocket socket) {
060    super(socket, socket.getLocalPort());
061
062    this.converter = new OSCByteArrayToJavaConverter();
063    this.dispatcher = new OSCPacketDispatcher();
064  }
065
066  /**
067   * Create an OSCPort that listens on the specified port.
068   * Strings will be decoded using the systems default character set.
069   * @param port UDP port to listen on.
070   * @throws SocketException if the port number is invalid,
071   *   or there is already a socket listening on it
072   */
073  public OSCPortIn(int port) throws SocketException {
074    this(new DatagramSocket(port));
075  }
076
077  /**
078   * Create an OSCPort that listens on the specified port,
079   * and decodes strings with a specific character set.
080   * @param port UDP port to listen on.
081   * @param charset how to decode strings read from incoming packages.
082   *   This includes message addresses and string parameters.
083   * @throws SocketException if the port number is invalid,
084   *   or there is already a socket listening on it
085   */
086  public OSCPortIn(int port, Charset charset) throws SocketException {
087    this(port);
088
089    this.converter.setCharset(charset);
090  }
091
092  /**
093   * Run the loop that listens for OSC on a socket until
094   * {@link #isListening()} becomes false.
095   * @see java.lang.Runnable#run()
096   */
097  @Override
098  public void run() {
099    final byte[] buffer = new byte[BUFFER_SIZE];
100    final DatagramPacket packet = new DatagramPacket(buffer, BUFFER_SIZE);
101    final DatagramSocket socket = getSocket();
102    while (listening) {
103      try {
104        try {
105          socket.receive(packet);
106        } catch (SocketException ex) {
107          if (listening) {
108            throw ex;
109          } else {
110            // if we closed the socket while receiving data,
111            // the exception is expected/normal, so we hide it
112            continue;
113          }
114        }
115        final OSCPacket oscPacket = converter.convert(buffer,
116            packet.getLength());
117        dispatcher.dispatchPacket(oscPacket);
118      } catch (IOException ex) {
119        ex.printStackTrace(); // XXX This may not be a good idea, as this could easily lead to a never ending series of exceptions thrown (due to the non-exited while loop), and because the user of the lib may want to handle this case himself
120      }
121    }
122  }
123
124  /**
125   * Start listening for incoming OSCPackets
126   */
127  public void startListening() {
128    listening = true;
129    final Thread thread = new Thread(this);
130    // The JVM exits when the only threads running are all daemon threads.
131    thread.setDaemon(true);
132    thread.start();
133  }
134
135  /**
136   * Stop listening for incoming OSCPackets
137   */
138  public void stopListening() {
139    listening = false;
140  }
141
142  /**
143   * Am I listening for packets?
144   * @return true if this port is in listening mode
145   */
146  public boolean isListening() {
147    return listening;
148  }
149
150  /**
151   * Registers a listener that will be notified of incoming messages,
152   * if their address matches the given pattern.
153   *
154   * @param addressSelector either a fixed address like "/sc/mixer/volume",
155   *   or a selector pattern (a mix between wildcards and regex)
156   *   like "/??/mixer/*", see {@link OSCPatternAddressSelector} for details
157   * @param listener will be notified of incoming packets, if they match
158   */
159  public void addListener(String addressSelector, OSCListener listener) {
160    this.addListener(new OSCPatternAddressSelector(addressSelector), listener);
161  }
162
163  /**
164   * Registers a listener that will be notified of incoming messages,
165   * if their address matches the given selector.
166   * @param addressSelector a custom address selector
167   * @param listener will be notified of incoming packets, if they match
168   */
169  public void addListener(AddressSelector addressSelector, OSCListener listener) {
170    dispatcher.addListener(addressSelector, listener);
171  }
172}
