009package com.illposed.osc;

011/**
012 * An OSC 1.0 optional, and OSC 1.1 required argument type.
013 * Impulse aka "bang", is used for event triggers.
014 * No bytes are allocated in the argument data.
015 * This type was named "Infinitum" in OSC 1.0.
016 * Use like this:
017 * <blockquote><pre>{@code
018 * OSCMessage msg = new OSCMessage("/my/address");
019 * msg.addArgument(OSCImpulse.INSTANCE);
020 * }</pre></blockquote>
021 */
022public final class OSCImpulse {
023
024  public static final OSCImpulse INSTANCE = new OSCImpulse();
025
026  private OSCImpulse() {}
027}
