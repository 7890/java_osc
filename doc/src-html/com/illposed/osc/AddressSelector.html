009package com.illposed.osc;
010
011/**
012 * Checks whether an OSC <i>Address Pattern</i> fulfills certain criteria.
013 */
014public interface AddressSelector {
015
016  /**
017   * Checks whether the OSC <i>Address Pattern</i> in question
018   * matches this selector.
019   * @param messageAddress for example "/sc/mixer/volume"
020   * @return true if this matcher selects the message address in question.
021   */
022  boolean matches(String messageAddress);
023}
