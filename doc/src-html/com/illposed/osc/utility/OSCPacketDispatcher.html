009package com.illposed.osc.utility;
010
011import com.illposed.osc.AddressSelector;
012import com.illposed.osc.OSCBundle;
013import com.illposed.osc.OSCListener;
014import com.illposed.osc.OSCMessage;
015import com.illposed.osc.OSCPacket;
016import java.util.Date;
017import java.util.HashMap;
018import java.util.List;
019import java.util.Map;
020import java.util.Map.Entry;
021
022/**
023 * Dispatches {@link OSCPacket}s to registered listeners (<i>Method</i>s).
024 *
025 * @author Chandrasekhar Ramakrishnan
026 */
027public class OSCPacketDispatcher {
028
029  private final Map<AddressSelector, OSCListener> selectorToListener;
030
031  public OSCPacketDispatcher() {
032    this.selectorToListener = new HashMap<AddressSelector, OSCListener>();
033  }
034
035  /**
036   * Adds a listener (<i>Method</i> in OSC speak) that will be notified
037   * of incoming messages that match the selector.
038   * @param addressSelector selects which messages will be forwarded to the listener,
039   *   depending on the message address
040   * @param listener receives messages accepted by the selector
041   */
042  public void addListener(AddressSelector addressSelector, OSCListener listener) {
043    selectorToListener.put(addressSelector, listener);
044  }
045
046  public void dispatchPacket(OSCPacket packet) {
047    dispatchPacket(packet, null);
048  }
049
050  public void dispatchPacket(OSCPacket packet, Date timestamp) {
051    if (packet instanceof OSCBundle) {
052      dispatchBundle((OSCBundle) packet);
053    } else {
054      dispatchMessage((OSCMessage) packet, timestamp);
055    }
056  }
057
058  private void dispatchBundle(OSCBundle bundle) {
059    final Date timestamp = bundle.getTimestamp();
060    final List<OSCPacket> packets = bundle.getPackets();
061    for (final OSCPacket packet : packets) {
062      dispatchPacket(packet, timestamp);
063    }
064  }
065
066  private void dispatchMessage(OSCMessage message, Date time) {
067    for (final Entry<AddressSelector, OSCListener> addrList : selectorToListener.entrySet()) {
068      if (addrList.getKey().matches(message.getAddress())) {
069        addrList.getValue().acceptMessage(time, message);
070      }
071    }
072  }
073}
