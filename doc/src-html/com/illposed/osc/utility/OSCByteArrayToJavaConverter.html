021/**
022 * Utility class to convert a byte array,
023 * conforming to the OSC byte stream format,
024 * into Java objects.
025 *
026 * @author Chandrasekhar Ramakrishnan
027 */
028public class OSCByteArrayToJavaConverter {
029
030  private static final String BUNDLE_START = "#bundle";
031  private static final char BUNDLE_IDENTIFIER = BUNDLE_START.charAt(0);
032  private static final String NO_ARGUMENT_TYPES = "";
033
034  private static class Input {
035
036    private final byte[] bytes;
037    private final int bytesLength;
038    private int streamPosition;
039
040    Input(final byte[] bytes, final int bytesLength) {
041
042      this.bytes = bytes;
043      this.bytesLength = bytesLength;
044      this.streamPosition = 0;
045    }
046
047    public byte[] getBytes() {
048      return bytes;
049    }
050
051    public int getBytesLength() {
052      return bytesLength;
053    }
054
055    public int getAndIncreaseStreamPositionByOne() {
056      return streamPosition++;
057    }
058
059    public void addToStreamPosition(int toAdd) {
060      streamPosition += toAdd;
061    }
062
063    public int getStreamPosition() {
064      return streamPosition;
065    }

098  /**
099   * Converts a byte array into an {@link OSCPacket}
100   * (either an {@link OSCMessage} or {@link OSCBundle}).
101   * @param bytes the storage containing the raw OSC packet
102   * @param bytesLength indicates how many bytes the package consists of (<code>&lt;= bytes.length</code>)
103   * @return the successfully parsed OSC packet; in case of a problem,
104   *   a <code>RuntimeException</code> is thrown
105   */
106  public OSCPacket convert(byte[] bytes, int bytesLength) {
107
108    final Input rawInput = new Input(bytes, bytesLength);
109    final OSCPacket packet;
110    if (isBundle(rawInput)) {
111      packet = convertBundle(rawInput);
112    } else {
113      packet = convertMessage(rawInput);
114    }
115
116    return packet;
117  }
118
119  /**
120   * Checks whether my byte array is a bundle.
121   * From the OSC 1.0 specifications:
122   * <quote>
123   * The contents of an OSC packet must be either an OSC Message
124   * or an OSC Bundle. The first byte of the packet's contents unambiguously
125   * distinguishes between these two alternatives.
126   * </quote>
127   * @return true if it the byte array is a bundle, false o.w.
128   */
129  private boolean isBundle(final Input rawInput) {
130    // The shortest valid packet may be no shorter then 4 bytes,
131    // thus we may assume to always have a byte at index 0.
132    return rawInput.getBytes()[0] == BUNDLE_IDENTIFIER;
133  }
134
135  /**
136   * Converts the byte array to a bundle.
137   * Assumes that the byte array is a bundle.
138   * @return a bundle containing the data specified in the byte stream
139   */
140  private OSCBundle convertBundle(final Input rawInput) {
141    // skip the "#bundle " stuff
142    rawInput.addToStreamPosition(BUNDLE_START.length() + 1);
143    final Date timestamp = readTimeTag(rawInput);
144    final OSCBundle bundle = new OSCBundle(timestamp);
145    final OSCByteArrayToJavaConverter myConverter
146        = new OSCByteArrayToJavaConverter();
147    myConverter.setCharset(charset);
148    while (rawInput.getStreamPosition() < rawInput.getBytesLength()) {
149      // recursively read through the stream and convert packets you find
150      final int packetLength = readInteger(rawInput);
151      if (packetLength == 0) {
152        throw new IllegalArgumentException("Packet length may not be 0");
153      } else if ((packetLength % 4) != 0) {
154        throw new IllegalArgumentException("Packet length has to be a multiple of 4, is:"
155            + packetLength);
156      }
157      final byte[] packetBytes = new byte[packetLength];
158      System.arraycopy(rawInput.getBytes(), rawInput.getStreamPosition(), packetBytes, 0, packetLength);
159      rawInput.addToStreamPosition(packetLength);
160      final OSCPacket packet = myConverter.convert(packetBytes, packetLength);
161      bundle.addPacket(packet);
162    }
163    return bundle;
164  }
165
166  /**
167   * Converts the byte array to a simple message.
168   * Assumes that the byte array is a message.
169   * @return a message containing the data specified in the byte stream
170   */
171  private OSCMessage convertMessage(final Input rawInput) {
172    final OSCMessage message = new OSCMessage();
173    message.setAddress(readString(rawInput));
174    final CharSequence types = readTypes(rawInput);
175    for (int ti = 0; ti < types.length(); ++ti) {
176      if ('[' == types.charAt(ti)) {
177        // we're looking at an array -- read it in
178        message.addArgument(readArray(rawInput, types, ++ti));
179        // then increment i to the end of the array
180        while (types.charAt(ti) != ']') {
181          ti++;
182        }
183      } else {
184        message.addArgument(readArgument(rawInput, types.charAt(ti)));
185      }
186    }
187    return message;
188  }
189
190  /**
191   * Reads a string from the byte stream.
192   * @return the next string in the byte stream
193   */
194  private String readString(final Input rawInput) {
195    final int strLen = lengthOfCurrentString(rawInput);
196    final String res = new String(rawInput.getBytes(), rawInput.getStreamPosition(), strLen, charset);
197    rawInput.addToStreamPosition(strLen);
198    moveToFourByteBoundry(rawInput);
199    return res;
200  }
201
202  /**
203   * Reads a binary blob from the byte stream.
204   * @return the next blob in the byte stream
205   */
206  private byte[] readBlob(final Input rawInput) {
207    final int blobLen = readInteger(rawInput);
208    final byte[] res = new byte[blobLen];
209    System.arraycopy(rawInput.getBytes(), rawInput.getStreamPosition(), res, 0, blobLen);
210    rawInput.addToStreamPosition(blobLen);
211    moveToFourByteBoundry(rawInput);
212    return res;
213  }

215  /**
216   * Reads the types of the arguments from the byte stream.
217   * @return a char array with the types of the arguments,
218   *   or <code>null</code>, in case of no arguments
219   */
220  private CharSequence readTypes(final Input rawInput) {
221    final String typesStr;
222
223    // The next byte should be a ',', but some legacy code may omit it
224    // in case of no arguments, refering to "OSC Messages" in:
225    // http://opensoundcontrol.org/spec-1_0
226    if (rawInput.getBytes().length <= rawInput.getStreamPosition()) {
227      typesStr = NO_ARGUMENT_TYPES;
228    } else if (rawInput.getBytes()[rawInput.getStreamPosition()] != ',') {
229      // XXX should we not rather fail-fast -> throw exception?
230      typesStr = NO_ARGUMENT_TYPES;
231    } else {
232      rawInput.getAndIncreaseStreamPositionByOne();
233      typesStr = readString(rawInput);
234    }
235
236    return typesStr;
237  }
238
239  /**
240   * Reads an object of the type specified by the type char.
241   * @param type type of the argument to read
242   * @return a Java representation of the argument
243   */
244  private Object readArgument(final Input rawInput, final char type) {
245    switch (type) {
246      case 'u' :
247        return readUnsignedInteger(rawInput);
248      case 'i' :
249        return readInteger(rawInput);
250      case 'h' :
251        return readLong(rawInput);
252      case 'f' :
253        return readFloat(rawInput);
254      case 'd' :
255        return readDouble(rawInput);
256      case 's' :
257        return readString(rawInput);
258      case 'b' :
259        return readBlob(rawInput);
260      case 'c' :
261        return readChar(rawInput);
262      case 'N' :
263        return null;
264      case 'T' :
265        return Boolean.TRUE;
266      case 'F' :
267        return Boolean.FALSE;
268      case 'I' :
269        return OSCImpulse.INSTANCE;
270      case 't' :
271        return readTimeTag(rawInput);
272      default:
273        // XXX Maybe we should let the user choose what to do in this
274        //   case (we encountered an unknown argument type in an
275        //   incomming message):
276        //   just ignore (return null), or throw an exception?
277//        throw new UnsupportedOperationException(
278//            "Invalid or not yet supported OSC type: '" + type + "'");
279        return null;
280    }
281  }

283  /**
284   * Reads a char from the byte stream.
285   * @return a {@link Character}
286   */
287  private Character readChar(final Input rawInput) {
288    return (char) rawInput.getBytes()[rawInput.getAndIncreaseStreamPositionByOne()];
289  }

291  private BigInteger readBigInteger(final Input rawInput, final int numBytes) {
292    final byte[] myBytes = new byte[numBytes];
293    System.arraycopy(rawInput.getBytes(), rawInput.getStreamPosition(), myBytes, 0, numBytes);
294    rawInput.addToStreamPosition(numBytes);
295    return  new BigInteger(myBytes);
296  }

298  /**
299   * Reads a double from the byte stream.
300   * @return a 64bit precision floating point value
301   */
302  private Object readDouble(final Input rawInput) {
303    final BigInteger doubleBits = readBigInteger(rawInput, 8);
304    return Double.longBitsToDouble(doubleBits.longValue());
305  }
306
307  /**
308   * Reads a float from the byte stream.
309   * @return a 32bit precision floating point value
310   */
311  private Float readFloat(final Input rawInput) {
312    final BigInteger floatBits = readBigInteger(rawInput, 4);
313    return Float.intBitsToFloat(floatBits.intValue());
314  }

316  /**
317   * Reads a double precision integer (64 bit integer) from the byte stream.
318   * @return double precision integer (64 bit)
319   */
320  private Long readLong(final Input rawInput) {
321    final BigInteger longintBytes = readBigInteger(rawInput, 8);
322    return longintBytes.longValue();
323  }

325  /**
326   * Reads an Integer (32 bit integer) from the byte stream.
327   * @return an {@link Integer}
328   */
329  private Integer readInteger(final Input rawInput) {
330    final BigInteger intBits = readBigInteger(rawInput, 4);
331    return intBits.intValue();
332  }

334  /**
335   * Reads an unsigned integer (32 bit) from the byte stream.
336   * This code is copied from {@see http://darksleep.com/player/JavaAndUnsignedTypes.html},
337   * which is licensed under the Public Domain.
338   * @return single precision, unsigned integer (32 bit) wrapped in a 64 bit integer (long)
339   */
340  private Long readUnsignedInteger(final Input rawInput) {
341
342    final int firstByte = (0x000000FF & ((int) rawInput.getBytes()[rawInput.getAndIncreaseStreamPositionByOne()]));
343    final int secondByte = (0x000000FF & ((int) rawInput.getBytes()[rawInput.getAndIncreaseStreamPositionByOne()]));
344    final int thirdByte = (0x000000FF & ((int) rawInput.getBytes()[rawInput.getAndIncreaseStreamPositionByOne()]));
345    final int fourthByte = (0x000000FF & ((int) rawInput.getBytes()[rawInput.getAndIncreaseStreamPositionByOne()]));
346    return ((long) (firstByte << 24
347        | secondByte << 16
348        | thirdByte << 8
349        | fourthByte))
350        & 0xFFFFFFFFL;
351  }

353  /**
354   * Reads the time tag and convert it to a Java Date object.
355   * A timestamp is a 64 bit number representing the time in NTP format.
356   * The first 32 bits are seconds since 1900, the second 32 bits are
357   * fractions of a second.
358   * @return a {@link Date}
359   */
360  private Date readTimeTag(final Input rawInput) {
361    final byte[] secondBytes = new byte[8];
362    final byte[] fractionBytes = new byte[8];
363    for (int bi = 0; bi < 4; bi++) {
364      // clear the higher order 4 bytes
365      secondBytes[bi] = 0;
366      fractionBytes[bi] = 0;
367    }
368    // while reading in the seconds & fraction, check if
369    // this timetag has immediate semantics
370    boolean isImmediate = true;
371    for (int bi = 4; bi < 8; bi++) {
372      secondBytes[bi] = rawInput.getBytes()[rawInput.getAndIncreaseStreamPositionByOne()];
373      if (secondBytes[bi] > 0) {
374        isImmediate = false;
375      }
376    }
377    for (int bi = 4; bi < 8; bi++) {
378      fractionBytes[bi] = rawInput.getBytes()[rawInput.getAndIncreaseStreamPositionByOne()];
379      if (bi < 7) {
380        if (fractionBytes[bi] > 0) {
381          isImmediate = false;
382        }
383      } else {
384        if (fractionBytes[bi] > 1) {
385          isImmediate = false;
386        }
387      }
388    }
389
390    if (isImmediate) {
391      return OSCBundle.TIMESTAMP_IMMEDIATE;
392    }
393
394    final long secsSince1900 = new BigInteger(secondBytes).longValue();
395    long secsSince1970 = secsSince1900 - OSCBundle.SECONDS_FROM_1900_TO_1970;
396
397    // no point maintaining times in the distant past
398    if (secsSince1970 < 0) {
399      secsSince1970 = 0;
400    }
401    long fraction = new BigInteger(fractionBytes).longValue();
402
403    // this line was cribbed from jakarta commons-net's NTP TimeStamp code
404    fraction = (fraction * 1000) / 0x100000000L;
405
406    // I do not know where, but I'm losing 1ms somewhere...
407    fraction = (fraction > 0) ? fraction + 1 : 0;
408    final long millisecs = (secsSince1970 * 1000) + fraction;
409    return new Date(millisecs);
410  }

412  /**
413   * Reads an array from the byte stream.
414   * @param types
415   * @param pos at which position to start reading
416   * @return the array that was read
417   */
418  private List<Object> readArray(final Input rawInput, final CharSequence types, int pos) {
419    int arrayLen = 0;
420    while (types.charAt(pos + arrayLen) != ']') {
421      arrayLen++;
422    }
423    final List<Object> array = new ArrayList<Object>(arrayLen);
424    for (int ai = 0; ai < arrayLen; ai++) {
425      array.add(readArgument(rawInput, types.charAt(pos + ai)));
426    }
427    return array;
428  }

430  /**
431   * Get the length of the string currently in the byte stream.
432   */
433  private int lengthOfCurrentString(final Input rawInput) {
434    int len = 0;
435    while (rawInput.getBytes()[rawInput.getStreamPosition() + len] != 0) {
436      len++;
437    }
438    return len;
439  }

441  /**
442   * Move to the next byte with an index in the byte array
443   * which is dividable by four.
444   */
445  private void moveToFourByteBoundry(final Input rawInput) {
446    // If i am already at a 4 byte boundry, I need to move to the next one
447    final int mod = rawInput.getStreamPosition() % 4;
448    rawInput.addToStreamPosition(4 - mod);
449  }
